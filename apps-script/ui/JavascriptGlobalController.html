class GlobalController { // SECTION: // Elements static getUpdateOverviewElement() { return $('.updateOverview'); } static getRerenderSidebarElement() { return $('.rerenderSidebar'); } static getDoAllTheThingsElement() { return $('.doAllTheThings'); } // SECTION: // Attributes static getSpreadsheetUrl() { return $('[data-spreadsheet-url]').attr('data-spreadsheet-url'); } // SECTION: // Click handlers static updateOverview() { new RpcConfig(runner => runner.updateOverview(GlobalController.getSpreadsheetUrl())). output('Updating overview...'). retryOnTimeout(). callRpc(); } static rerenderSidebar() { new RpcConfig(runner => runner.showPreferencesSidebar(GlobalController.getSpreadsheetUrl())).output('Rerendering sidebar...').callRpc(); } static doAllTheThings() { SheetController.doYourThings().then(GlobalController.updateOverview); } static addListeners() { GlobalController.getUpdateOverviewElement().click(GlobalController.updateOverview); GlobalController.getRerenderSidebarElement().click(GlobalController.rerenderSidebar); GlobalController.getDoAllTheThingsElement().click(GlobalController.doAllTheThings); } static init() { GlobalController.addListeners(); } }
