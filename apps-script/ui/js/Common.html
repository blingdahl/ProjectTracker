class Common { static textFn(emptyText) { return text => text || emptyText; } static toDateStr(d) { let monthStr = String(d.getMonth() + 1); if (monthStr.length === 1) { monthStr = '0' + monthStr; } let dayStr = String(d.getDate()); if (dayStr.length === 1) { dayStr = '0' + dayStr; } return (d.getYear() + 1900) + '/' + monthStr + '/' + dayStr; } static getCurrDateStr() { return Common.toDateStr(new Date()); } static getNextDateStr() { let d = new Date(); d.setDate(d.getDate() + 1); return Common.toDateStr(d); } static typeOf(value) { var s = typeof value; if (s === 'object') { if (value) { if (value instanceof Array) { s = 'array'; } } else { s = 'null'; } } return s; } static toArray(val) { if (Common.typeOf(val) == 'array') { return val; } return [val]; } static ensureProtocol(href) { if (href.startsWith('http://') || href.startsWith('https://')) { return href; } return 'http://' + href; } }
