class Common { static textFn(emptyText) { return text => text || emptyText; } static typeOf(value) { var s = typeof value; if (s === 'object') { if (value) { if (value instanceof Array) { s = 'array'; } } else { s = 'null'; } } return s; } static toArray(val) { if (Common.typeOf(val) == 'array') { return val; } return [val]; } static ensureProtocol(href) { if (href.startsWith('http://') || href.startsWith('https://')) { return href; } return 'http://' + href; } }
