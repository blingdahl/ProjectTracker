class OptionInputMaker { constructor(allValues, emptyText, cancelCallback) { this.allValues = allValues; this.emptyText = emptyText; this.cancelCallback = cancelCallback; } makeElement(currValue, changeCallback, cancelCallback) { let optionsJq = $('
').addClass('optionPicker'); for (var i = 0; i != this.allValues.length; i++) { let optionValue = this.allValues[i]; let text; let value = optionValue; if (typeof optionValue === 'string') { text = optionValue || this.emptyText; value = optionValue; } else { text = optionValue.text; value = optionValue.value; } let optionJq = $('
').attr('data-value', value).attr('data-text', text).text(text).addClass('option'); if (currValue == value) { optionJq.addClass('currValue'); } optionsJq.append(optionJq); optionJq.click((event) => { let jq = $(event.currentTarget); let newValue = jq.attr('data-value'); let newText = jq.text(); optionsJq.remove(); if (newValue != currValue) { changeCallback(newValue, newText); } }); } optionsJq.append(new ButtonBar([ {text: 'Cancel', callback: () => { optionsJq.remove(); cancelCallback(); } }]).makeElement()); return optionsJq; } }
