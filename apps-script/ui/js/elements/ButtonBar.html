class ButtonBar { constructor(buttons) { this.buttons = buttons; } makeElement() { let buttonsJq = $('
').addClass('buttons'); for (let i = 0; i != this.buttons.length; i++) { if (i != 0) { buttonsJq.append($('').addClass('betweenButtons')); } let button = this.buttons[i]; buttonsJq.append($('').text(button.text).addClass('button').onClick(button.callback)); } return buttonsJq; } }
