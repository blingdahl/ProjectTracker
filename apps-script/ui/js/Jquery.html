/** * @fileoverview Jquery functions. */ jQuery.fn.extend({ onClick: function(callback) { return this.click(event => { callback(event); event.preventDefault(); event.stopPropagation(); }); }, onKeydown: function(callback) { this.keydown(event => { var keycode = (event.keyCode ? event.keyCode : event.which); if (callback(event, keycode)) { event.stopPropagation(); event.preventDefault(); } }); } });
