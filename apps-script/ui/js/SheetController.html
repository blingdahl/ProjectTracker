class SheetController { constructor(jq) { this.jq = jq; this.labelPicker = new LabelPicker(this.jq.find('.labelPicker'), this.getSheetId()); } // SECTION: // Elements getTrackElement() { return this.jq.find('.track'); } getUpdateElement() { return this.jq.find('.update'); } getAddElement() { return this.jq.find('.add'); } getTidyUpColumnsElement() { return this.jq.find('.tidyUpColumns'); } getAddTrackingElement() { return this.jq.find('.addTracking'); } getSearchElement() { return this.jq.find('.search'); } static getDoAllTheThingsElement() { return $('.doAllTheThings'); } static getTidyAllElement() { return $('.tidyAll'); } // SECTION: // Attributes getSheetId() { return parseInt(this.jq.attr('data-sheet-id')); } getSheetName() { return this.jq.attr('data-sheet-name'); } getSearchQuery() { return this.jq.attr('data-search-query'); } hide() { this.jq.hide(); } show() { let sheetName = this.getSheetName(); if (sheetName === 'Overview') { this.hide(); return; } this.labelPicker.setMode(LabelPicker.MODE.DISPLAY); this.jq.show(); } setAction(uuid, action) { return new RpcConfig(runner => runner.ChangeRowValue_setAction(this.getSheetId(), uuid, action)). output('Setting action to ' + action + '...'). retryOnTimeout(). callRpc(); } setPriority(uuid, priority) { return new RpcConfig(runner => runner.ChangeRowValue_setPriority(this.getSheetId(), uuid, priority)). output('Setting priority to ' + priority + '...'). retryOnTimeout(). callRpc(); } setStatus(uuid, status) { return new RpcConfig(runner => runner.ChangeRowValue_setStatus(this.getSheetId(), uuid, status)). output('Setting status to ' + status + '...'). retryOnTimeout(). callRpc(); } // SECTION: // Click handlers update() { return new RpcConfig(runner => runner.Update_update(this.getSheetId())). output('Updating ' + this.getSheetName() + ' sheet...'). retryOnTimeout(). callRpc(). then(() => OverviewController.updateOverviewForSheet(this.getSheetId())); } add() { new AddItemDialog(this.getSheetId(), this.getSheetName(), GlobalController.PRIORITIES).show(); } tidyUpColumns() { new RpcConfig(runner => runner.TidyUpColumns_tidyUpColumns(this.getSheetId())). output('Tidying up ' + this.getSheetName() + '...'). retryOnTimeout(). callRpc(); } addTracking() { new RpcConfig(runner => runner.Preferences_setTrackedForSheet(this.getSheetId(), true)). output('Adding tracking for ' + this.getSheetName() + ' sheet...'). retryOnTimeout(). callRpc(); } search() { window.open('https://mail.google.com/mail/u/0/#search/' + this.getSearchQuery()); } static updateAll() { SheetController.getAll().forEach(sheetController => sheetController.update()); } static tidyAll() { SheetController.getAll().forEach(sheetController => sheetController.tidyUpColumns()); } // SECTION: // Initialization addListeners() { this.labelPicker.addListeners(); this.getUpdateElement().click(this.update.bind(this)); this.getAddElement().click(this.add.bind(this)); this.getAddTrackingElement().click(this.addTracking.bind(this)); this.getSearchElement().click(this.search.bind(this)); } static addGlobalListeners() { SheetController.getDoAllTheThingsElement().click(SheetController.updateAll); SheetController.getTidyAllElement().click(SheetController.tidyAll); } static getAll() { return SheetController.ALL; } static getSheetInfos() { var allSheetControllers = SheetController.getAll(); var sheetInfos = []; allSheetControllers.forEach(sheetController => sheetInfos.push({sheetId: sheetController.getSheetId(), sheetName: sheetController.getSheetName()})); return sheetInfos; } static forSheet(sheetId) { let ret = null; SheetController.ALL.forEach(sheetController => { if (sheetController.getSheetId() == sheetId) { ret = sheetController; } }); return ret; } static init() { $('.sheetController').each(function() { let sheetController = new SheetController($(this)); sheetController.addListeners(); sheetController.show(); SheetController.ALL.push(sheetController); }); SheetController.addGlobalListeners(); } } SheetController.ALL = [];
