class SheetController { constructor(jq) { this.jq = jq; this.labelPicker = new LabelPicker(this.jq.find('.labelPicker'), this.getSheetId()); } // SECTION: // Elements getTrackElement() { return this.jq.find('.track'); } getUpdateElement() { return this.jq.find('.update'); } getAddElement() { return this.jq.find('.add'); } getTidyUpColumnsElement() { return this.jq.find('.tidyUpColumns'); } getAddTrackingElement() { return this.jq.find('.addTracking'); } getSearchElement() { return this.jq.find('.search'); } static getDoAllTheThingsElement() { return $('.doAllTheThings'); } static getTidyAllElement() { return $('.tidyAll'); } static getUpdateDirtyElement() { return $('.updateDirty'); } // SECTION: // Attributes getSheetId() { return parseInt(this.jq.attr('data-sheet-id')); } getSheetName() { return this.jq.attr('data-sheet-name'); } getSearchQuery() { return this.jq.attr('data-search-query'); } isDirty() { return this.jq.attr('data-dirty') === 'true'; } markDirty() { this.jq.attr('data-dirty', 'true'); } markNotDirty() { this.jq.attr('data-dirty', null); } hide() { this.jq.hide(); } show() { let sheetName = this.getSheetName(); if (sheetName === 'Overview') { this.hide(); return; } this.labelPicker.setMode(LabelPicker.MODE.DISPLAY); this.jq.show(); } setValue(uuid, columnName, columnKey, newValue) { this.markDirty(); return new RpcConfig('ChangeRowValue_setValue', runner => runner.ChangeRowValue_setValue(this.getSheetId(), uuid, columnName, columnKey, newValue)). output('Setting ' + columnName + ' to ' + newValue + '...'). callRpc(); } setLink(uuid, text, href) { this.markDirty(); new RpcConfig('ChangeRowValue_setLink', runner => runner.ChangeRowValue_setLink(this.getSheetId(), uuid, 'Link', 'LINK', text, href)). output('Saving new link...'). callRpc(); } clearLink(uuid, text, href) { this.markDirty(); new RpcConfig('ChangeRowValue_setLink', runner => runner.ChangeRowValue_clearLink(this.getSheetId(), uuid, 'Link', 'LINK')). output('Clearing link...'). callRpc(); } changeSheet(uuid, toSheetId) { this.markDirty(); return new RpcConfig('ChangeRowValue_changeSheet', runner => runner.ChangeRowValue_changeSheet(this.getSheetId(), toSheetId, uuid)). output('Moving to ' + SheetController.forSheet(toSheetId).getSheetName() + '...'). callRpc(); } // SECTION: // Click handlers afterUpdate() { this.updateOutstanding = false; if (this.updateQueued) { this.updateQueued = false; return this.update(); } else { return OverviewController.updateOverviewForSheet(this.getSheetId()); } } update() { if (this.updateOutstanding) { this.updateQueued = true; return; } this.updateOutstanding = true; this.markNotDirty(); return new RpcConfig('Update_update', runner => runner.Update_update(this.getSheetId())). output('Updating ' + this.getSheetName() + ' sheet...'). callRpc(). then(this.afterUpdate.bind(this)); } addItemCallback() { this.markDirty(); } add() { new AddItemDialog(this.getSheetId(), this.getSheetName(), this.addItemCallback.bind(this)).show(); } tidyUpColumns() { new RpcConfig('TidyUpColumns_tidyUpColumns', runner => runner.TidyUpColumns_tidyUpColumns(this.getSheetId())). output('Tidying up ' + this.getSheetName() + '...'). callRpc(); } addTracking() { new RpcConfig('Preferences_setTrackedForSheet', runner => runner.Preferences_setTrackedForSheet(this.getSheetId(), true)). output('Adding tracking for ' + this.getSheetName() + ' sheet...'). callRpc(); } search() { window.open('https://mail.google.com/mail/u/0/#search/' + this.getSearchQuery()); } static updateAllDirty() { SheetController.getAll().forEach(sheetController => { if (sheetController.isDirty()) { sheetController.update(); } }); } static updateAll() { SheetController.getAll().forEach(sheetController => sheetController.update()); } static tidyAll() { SheetController.getAll().forEach(sheetController => sheetController.tidyUpColumns()); } // SECTION: // Initialization addListeners() { this.labelPicker.addListeners(); this.getUpdateElement().click(this.update.bind(this)); this.getAddElement().click(this.add.bind(this)); this.getAddTrackingElement().click(this.addTracking.bind(this)); this.getSearchElement().click(this.search.bind(this)); } static addGlobalListeners() { SheetController.getDoAllTheThingsElement().click(SheetController.updateAll); SheetController.getTidyAllElement().click(SheetController.tidyAll); SheetController.getUpdateDirtyElement().click(SheetController.updateAllDirty); $(document).keydown((event) => { var keycode = (event.keyCode ? event.keyCode : event.which); if (keycode === 13 && event.metaKey) { SheetController.updateAll(); event.stopPropagation(); event.preventDefault(); } if (keycode === 68 && event.metaKey) { SheetController.updateAllDirty(); event.stopPropagation(); event.preventDefault(); } }); } static getAll() { return SheetController.ALL; } static getSheetInfos() { if (!SheetController.SHEET_INFOS) { var allSheetControllers = SheetController.getAll(); SheetController.SHEET_INFOS = []; allSheetControllers.forEach(sheetController => SheetController.SHEET_INFOS.push({sheetId: sheetController.getSheetId(), sheetName: sheetController.getSheetName()})); SheetController.SHEET_INFOS.sort((a, b) => a.sheetName > b.sheetName ? 1 : a.sheetName != b.sheetName ? -1 : 0); } return SheetController.SHEET_INFOS; } static getSheetIds() { let sheetIds = []; this.getSheetInfos().forEach(sheetInfo => sheetIds.push(sheetInfo.sheetId)); return sheetIds; } static getSheetName(sheetId) { return SheetController.forSheet(sheetId).getSheetName(); } static forSheet(sheetId) { let ret = null; SheetController.ALL.forEach(sheetController => { if (sheetController.getSheetId() == sheetId) { ret = sheetController; } }); return ret; } static init() { $('.sheetController').each(function() { let sheetController = new SheetController($(this)); sheetController.addListeners(); sheetController.show(); SheetController.ALL.push(sheetController); }); SheetController.addGlobalListeners(); } } SheetController.ALL = []; SheetController.SHEET_INFOS = null;
