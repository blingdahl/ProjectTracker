class LinkMaker { constructor(cssClass, textCallback, urlCallback, opt_emptyCallback) { this.cssClass = cssClass; this.textCallback = textCallback; this.urlCallback = urlCallback; this.opt_emptyCallback = opt_emptyCallback; } // Called once per table load attachClickHandler(ancestorJq) { // No-op } makeElement(data) { let url = this.urlCallback(data); if (!url) { if (this.opt_emptyCallback) { return this.opt_emptyCallback(data); } return null; } return $('').attr('href', url).addClass(this.cssClass).attr('target', '_blank').text(this.textCallback(data)); } }
