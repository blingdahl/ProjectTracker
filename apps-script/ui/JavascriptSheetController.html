class SheetController { constructor(jq) { this.jq = jq; this.labelPicker = new LabelPicker(this.jq.find('.labelPicker'), this.getSheetId()); } // SECTION: // Elements getTrackElement() { return this.jq.find('.track'); } getSyncWithGmailElement() { return this.jq.find('.syncWithGmail'); } getOrganizeElement() { return this.jq.find('.organize'); } getAddUrlElement() { return this.jq.find('.addUrl'); } getTidyUpColumnsElement() { return this.jq.find('.tidyUpColumns'); } getAddTrackingElement() { return this.jq.find('.addTracking'); } getSearchElement() { return this.jq.find('.search'); } static getDoAllTheThingsElement() { return $('.doAllTheThings'); } static getTidyAllElement() { return $('.tidyAll'); } // SECTION: // Attributes getSheetId() { return parseInt(this.jq.attr('data-sheet-id')); } getSheetName() { return this.jq.attr('data-sheet-name'); } getSearchQuery() { return this.jq.attr('data-search-query'); } hide() { this.jq.hide(); } show() { let sheetName = this.getSheetName(); if (sheetName === 'Overview') { this.hide(); return; } this.labelPicker.setMode(LabelPicker.MODE.DISPLAY); this.jq.show(); } setAction(uuid, action) { return new RpcConfig(runner => runner.setAction(GlobalController.getSpreadsheetUrl(), this.getSheetId(), uuid, action)). output('Setting action to ' + action + '...'). retryOnTimeout(). callRpc(); } // SECTION: // Click handlers organize() { return new RpcConfig(runner => runner.organize(GlobalController.getSpreadsheetUrl(), this.getSheetId())). output('Organizing ' + this.getSheetName() + ' sheet...'). retryOnTimeout(). callRpc(); } syncWithGmail() { if (!this.labelPicker.getLabelName()) { return Promise.resolve(); } return new RpcConfig(runner => runner.syncSheetWithGmail(GlobalController.getSpreadsheetUrl(), this.getSheetId())). output('Syncing ' + this.getSheetName() + ' sheet with ' + this.labelPicker.getLabelName() + ' label...'). retryOnTimeout(). callRpc(); } doYourThing() { if (this.labelPicker.getLabelName()) { return this.syncWithGmail(); } else { return this.organize(); } } addUrl() { let url = window.prompt('URL'); if (url) { new RpcConfig(runner => runner.addUrl(GlobalController.getSpreadsheetUrl(), this.getSheetId(), url)). output('Adding ' + url + ' to ' + this.getSheetName() + '...'). retryOnTimeout(). callRpc(); } } tidyUpColumns() { new RpcConfig(runner => runner.tidyUpColumns(GlobalController.getSpreadsheetUrl(), this.getSheetId())). output('Tidying up ' + this.getSheetName() + '...'). retryOnTimeout(). callRpc(); } addTracking() { new RpcConfig(runner => runner.setTrackedForSheet(GlobalController.getSpreadsheetUrl(), this.getSheetId(), true)). output('Adding tracking for ' + this.getSheetName() + ' sheet...'). retryOnTimeout(). callRpc(); } search() { window.open('https://mail.google.com/mail/u/0/#search/' + this.getSearchQuery()); } static doYourThings() { let promises = []; SheetController.getAll().forEach(sheetController => promises.push(sheetController.doYourThing())); return Promise.all(promises); } static tidyAll() { let promises = []; SheetController.getAll().forEach(sheetController => promises.push(sheetController.tidyUpColumns())); return Promise.all(promises); } // SECTION: // Initialization addListeners() { this.labelPicker.addListeners(); this.getSyncWithGmailElement().click(this.syncWithGmail.bind(this)); this.getOrganizeElement().click(this.organize.bind(this)); this.getAddUrlElement().click(this.addUrl.bind(this)); this.getAddTrackingElement().click(this.addTracking.bind(this)); this.getSearchElement().click(this.search.bind(this)); } static addGlobalListeners() { SheetController.getDoAllTheThingsElement().click(SheetController.doYourThings); SheetController.getTidyAllElement().click(SheetController.tidyAll); } static getAll() { return SheetController.ALL; } static forSheet(sheetId) { let ret = null; SheetController.ALL.forEach(sheetController => { if (sheetController.getSheetId() === sheetId) { ret = sheetController; } }); return ret; } static init() { $('.sheetController').each(function() { let sheetController = new SheetController($(this)); sheetController.addListeners(); sheetController.show(); SheetController.ALL.push(sheetController); }); SheetController.addGlobalListeners(); } } SheetController.ALL = [];
