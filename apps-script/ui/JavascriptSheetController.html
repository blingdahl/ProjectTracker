class SheetController { constructor(divJq) { this.divJq = divJq; this.labelPicker = new LabelPicker(this.divJq.find('.labelPicker'), this.getSheetId()); } // SECTION: // Elements getTrackElement() { return this.divJq.find('.track'); } getSyncWithGmailElement() { return this.divJq.find('.syncWithGmail'); } getOrganizeElement() { return this.divJq.find('.organize'); } getAddUrlElement() { return this.divJq.find('.addUrl'); } getAddTrackingElement() { return this.divJq.find('.addTracking'); } static getSyncAllWithGmailElement() { return $('.syncAllWithGmail'); } static getOrganizeAllElement() { return $('.organizeAll'); } // SECTION: // Attributes getSheetId() { return this.divJq.attr('data-sheet-id'); } getSheetName() { return this.divJq.attr('data-sheet-name'); } hide() { this.divJq.hide(); } updateTemplatedText(sheetName) { this.divJq.find('[data-templated-name]').each(function() { let jq = $(this); if (sheetName) { jq.text(jq.attr('data-templated-name').replace(/{sheetName}/g, sheetName)); } else { jq.text(jq.attr('data-no-templated-name')); } }); } show() { let sheetName = this.getSheetName(); if (sheetName === 'Overview') { this.hide(); return; } this.updateTemplatedText(sheetName); this.labelPicker.setMode(LabelPicker.MODE.DISPLAY); this.divJq.show(); } // SECTION: // Click handlers organize() { return new RpcConfig(runner => runner.organize(GlobalController.getSpreadsheetUrl(), this.getSheetId())). output('Organizing ' + this.getSheetName() + ' sheet...'). retryOnTimeout(). callRpc(); } syncWithGmail() { if (!this.labelPicker.getLabelName()) { return Promise.resolve(); } return new RpcConfig(runner => runner.syncSheetWithGmail(GlobalController.getSpreadsheetUrl(), this.getSheetId())). output('Syncing ' + this.getSheetName() + ' sheet with ' + this.labelPicker.getLabelName() + ' label...'). retryOnTimeout(). callRpc(); } doYourThing() { if (this.labelPicker.getLabelName()) { return this.syncWithGmail(); } else { return this.organize(); } } addUrl() { let url = window.prompt('URL'); if (url) { new RpcConfig(runner => runner.addUrl(GlobalController.getSpreadsheetUrl(), this.getSheetId(), url)). output('Adding ' + url + ' to ' + this.getSheetName() + '...'). retryOnTimeout(). callRpc(); } } addTracking() { new RpcConfig(runner => runner.setTrackedForSheet(GlobalController.getSpreadsheetUrl(), this.getSheetId(), true)). output('Adding tracking for ' + this.getSheetName() + ' sheet...'). retryOnTimeout(). callRpc(); } static organizeAll() { let sheetControllers = SheetController.getAll(); sheetControllers.forEach(sheetController => sheetController.organize()); } static syncAllWithGmail() { let sheetControllers = SheetController.getAll(); sheetControllers.forEach(sheetController => sheetController.syncWithGmail()); } static doYourThings() { let promises = []; SheetController.getAll().forEach(sheetController => promises.push(sheetController.doYourThing())); return Promise.all(promises); } // SECTION: // Initialization addListeners() { this.labelPicker.addListeners(); this.getSyncWithGmailElement().click(this.syncWithGmail.bind(this)); this.getOrganizeElement().click(this.organize.bind(this)); this.getAddUrlElement().click(this.addUrl.bind(this)); this.getAddTrackingElement().click(this.addTracking.bind(this)); } static addGlobalListeners() { SheetController.getOrganizeAllElement().click(SheetController.organizeAll); SheetController.getSyncAllWithGmailElement().click(SheetController.syncAllWithGmail); } static getAll() { return SheetController.ALL; } static init() { $('.sheetController').each(function() { let sheetController = new SheetController($(this)); sheetController.addListeners(); sheetController.show(); SheetController.ALL.push(sheetController); }); SheetController.addGlobalListeners(); } } SheetController.ALL = [];
