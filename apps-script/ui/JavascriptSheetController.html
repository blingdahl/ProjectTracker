class SheetController { constructor(divJq) { this.divJq = divJq; this.labelPicker = new LabelPicker(this.divJq.find('.labelPicker'), this.getSheetId()); } // SECTION: // Elements getTrackElement() { return this.divJq.find('.track'); } getSyncWithGmailElement() { return this.divJq.find('.syncWithGmail'); } getOrganizeElement() { return this.divJq.find('.organize'); } getAddUrlElement() { return this.divJq.find('.addUrl'); } getAddTrackingElement() { return this.divJq.find('.addTracking'); } // SECTION: // Attributes getSheetId() { return this.divJq.attr('data-sheet-id'); } getSheetName() { return this.divJq.attr('data-sheet-name'); } hide() { this.divJq.hide(); } updateTemplatedText(sheetName) { this.divJq.find('[data-templated-name]').each(function() { let jq = $(this); if (sheetName) { jq.text(jq.attr('data-templated-name').replace(/{sheetName}/g, sheetName)); } else { jq.text(jq.attr('data-no-templated-name')); } }); } show() { let sheetName = this.getSheetName(); if (sheetName === 'Overview') { this.hide(); return; } this.updateTemplatedText(sheetName); this.labelPicker.setMode(LabelPicker.MODE.DISPLAY); this.divJq.show(); } // SECTION: // Click handlers organizeClicked() { new RpcConfig(runner => runner.organize(GlobalController.getSpreadsheetUrl(), this.getSheetId())). output('Organizing ' + this.getSheetName() + ' sheet...'). retryOnTimeout(). callRpc(); } syncWithGmailClicked() { new RpcConfig(runner => runner.syncSheetWithGmail(GlobalController.getSpreadsheetUrl(), this.getSheetId())). output('Syncing ' + this.getSheetName() + ' sheet with ' + this.labelPicker.getLabelName() + ' label...'). retryOnTimeout(). callRpc(); } addUrlClicked() { let url = window.prompt('URL'); if (url) { new RpcConfig(runner => runner.addUrl(GlobalController.getSpreadsheetUrl(), this.getSheetId(), url)). output('Adding ' + url + ' to ' + this.getSheetName() + '...'). retryOnTimeout(). callRpc(); } } addTrackingClicked() { new RpcConfig(runner => runner.setTrackedForSheet(GlobalController.getSpreadsheetUrl(), this.getSheetId(), true)). output('Adding tracking for ' + this.getSheetName() + ' sheet...'). retryOnTimeout(). callRpc(); } // SECTION: // Initialization addListeners() { this.labelPicker.addListeners(); this.getSyncWithGmailElement().click(this.syncWithGmailClicked.bind(this)); this.getOrganizeElement().click(this.organizeClicked.bind(this)); this.getAddUrlElement().click(this.addUrlClicked.bind(this)); this.getAddTrackingElement().click(this.addTrackingClicked.bind(this)); } static init() { $('.sheetController').each(function() { let sheetController = new SheetController($(this)); sheetController.addListeners(); sheetController.show(); }); } }
